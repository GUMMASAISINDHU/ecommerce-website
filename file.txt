// Declare the branch parameter
properties([
    parameters([
        string(name: 'BRANCH_NAME', defaultValue: 'main', description: 'Branch to check out')
    ])
])

pipeline {
    agent any
    
    stages {
        stage('Checkout') {
            steps {
                // Checkout the specified branch
                checkout([
                    $class: 'GitSCM',
                    branches: [[name: params.BRANCH_NAME]], // Use the branch name from the parameter
                    userRemoteConfigs: [[
                        url: 'https://github.com/GUMMASAISINDHU/ecommerce-website.git',
                        credentialsId: 'your-credentials-id' // Specify credentials if needed
                    ]]
                ])
            }
        }
        
        stage('List Commits') {
            steps {
                script {
                    // Get the current commit ID and list the latest commits
                    def currentCommitID = env.GIT_COMMIT
                    println "Current Commit ID: $currentCommitID"
                    
                    // Fetch the latest commits using a git command within a shell
                    sh '''
                    git log --pretty=format:"%h - %s" -n 5
                    '''
                }
            }
        }
    }
}

pipeline { agent any stages { stage('Display Branches and Commit IDs') { steps { script { // Retrieve the list of branches from the Git repository def branches = sh(script: 'git ls-remote --heads origin', returnStdout: true).trim().split('\n').collect { it.replaceAll(/.*refs\/heads\//, '') } // Display the dropdown menu header echo "Branch\t\tCommit ID" echo "-----------------------------" // Iterate over each branch branches.each { branch -> // Checkout the branch checkout([$class: 'GitSCM', branches: [[name: branch]], userRemoteConfigs: [[url: 'https://github.com/GUMMASAISINDHU/ecommerce-website.git']]]) // Get the commit ID of the latest commit on the branch def commitID = sh(script: 'git rev-parse HEAD', returnStdout: true).trim() // Display the branch name and its commit ID echo "${branch}\t\t${commitID}" } } } } } }
